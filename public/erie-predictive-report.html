# linear model
from sklearn.linear_model import LinearRegression

# remove populations over 1 million to improve r-squared
query = "Population < 1000000"
population_and_providers = population_and_providers.query(query)

model = LinearRegression()
features = ["Population"]
X = population_and_providers[features]
label = ["NumProviders"]
y = population_and_providers[label]
model.fit(X, y)
r_squared = model.score(X, y)
print(r_squared)
(
    ggplot(data=population_and_providers)
    + aes(x="Population", y="NumProviders")
    + theme_bw()
    + geom_point()
    + geom_smooth(method="lm")
)
